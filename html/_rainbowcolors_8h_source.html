10  // Helper function to convert HSV (with h in degrees, s and v in [0,1])
11  // to a cv::Scalar in BGR order (each channel in [0,255]).
12  inline cv::Scalar HSVtoBGR(float h, float s, float v) {
13  float c = v * s;
14  float hPrime = h / 60.0f;
15  float x = c * (1 - fabsf(fmod(hPrime, 2.0f) - 1));
16  float r = 0.0f, g = 0.0f, b = 0.0f;
17 
18  if (0.0f <= hPrime && hPrime < 1.0f) {
19  r = c; g = x; b = 0;
20  }
21  else if (1.0f <= hPrime && hPrime < 2.0f) {
22  r = x; g = c; b = 0;
23  }
24  else if (2.0f <= hPrime && hPrime < 3.0f) {
25  r = 0; g = c; b = x;
26  }
27  else if (3.0f <= hPrime && hPrime < 4.0f) {
28  r = 0; g = x; b = c;
29  }
30  else if (4.0f <= hPrime && hPrime < 5.0f) {
31  r = x; g = 0; b = c;
32  }
33  else if (5.0f <= hPrime && hPrime < 6.0f) {
34  r = c; g = 0; b = x;
35  }
36 
37  float m = v - c;
38  r += m;
39  g += m;
40  b += m;
41 
42  // OpenCV uses BGR order.
43  return cv::Scalar(b * 255, g * 255, r * 255);
44  }

46  // Returns a constant array of 100 rainbow shades.
47  inline const std::array<cv::Scalar, 100>& getRainbowColors() {
48  static std::array<cv::Scalar, 100> rainbowColors;
49  static bool initialized = false;
50  if (!initialized) {
51  // Generate 100 colors spanning the hue range 0–360.
52  for (int i = 0; i < 100; ++i) {
53  // Linearly interpolate hue from 0 to 360 degrees.
54  float hue = (360.0f * i) / 100.0f;
55  rainbowColors[i] = HSVtoBGR(hue, 1.0f, 1.0f);
56  }
57  initialized = true;
58  }
59  return rainbowColors;
60  }
